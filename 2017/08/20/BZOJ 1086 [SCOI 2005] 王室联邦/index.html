void dfs(int step,int x)
{
vis[x]=true;int bot=top;
for (int q=first[x];q;q=nxt[q])
{
if (!vis[to[q]])
{
dfs(step+1,to[q]);

{
capital[++tot]=x;
while(top!=bot) belong[stack[top–]]=tot;
}
}
}
stack[++top]=x;
}
