inline double calc(double x)
{
    double temp=(hD-H*x)/(D-x);
    if (temp>0) return x+temp;
    else return (D-x)/(H-h);
}

        {
            double ll=(r-l)/3+l,rr=(r-l)*2/3+l;

            if (ansll>ansrr) r=rr;
